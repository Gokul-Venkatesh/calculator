-- Create stu_details table
CREATE TABLE stu_details (
    reg_no INT PRIMARY KEY,
    name VARCHAR(50),
    DOB DATE,
    address VARCHAR(100),
    city VARCHAR(50)
);

-- Create mark_details table
CREATE TABLE mark_details (
    reg_no INT PRIMARY KEY,
    mark1 INT,
    mark2 INT,
    mark3 INT,
    total INT
);

-- Display rows with total between 250 and 300
SELECT * FROM mark_details WHERE total BETWEEN 250 AND 300;

-- Drop mark_details table
DROP TABLE mark_details;

-- Delete row with reg_no = 161
DELETE FROM stu_details WHERE reg_no = 161;

-- Display details whose names begin with 'a'
SELECT * FROM stu_details WHERE name LIKE 'a%';

-- Control structure to print remarks on marks
DECLARE @total INT;
SET @total = (SELECT total FROM mark_details WHERE reg_no = <reg_no>); -- Replace <reg_no> with the desired registration number

IF @total >= 90 THEN
    PRINT 'Excellent!';
ELSEIF @total >= 80 THEN
    PRINT 'Very Good!';
ELSEIF @total >= 70 THEN
    PRINT 'Good!';
ELSEIF @total >= 60 THEN
    PRINT 'Satisfactory!';
ELSE
    PRINT 'Needs Improvement!';
END IF;
